public class BofP_lab1{
	private int[] arr;
	public void init(int size){
		arr = new int[size];
		for(int i=0;i<size;i++){
			arr[i] = i+1;
		}
	}
	public void printArray(){
		for(int i: arr)
			System.out.print(i+" ");
	} 
        public static void main(String [] args){
		BofP_lab1 b = new BofP_lab1();
		b.init(13);
		b.printArray();
        }
}
